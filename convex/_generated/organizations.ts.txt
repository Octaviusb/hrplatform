import { v } from "convex/values";
import { mutation, query } from "./_generated/server";
import { ConvexError } from "convex/values";

export const create = mutation({
  args: {
    name: v.string(),
    nit: v.string(),
    address: v.optional(v.string()),
    phone: v.optional(v.string()),
    email: v.optional(v.string()),
    industry: v.optional(v.string()),
    size: v.union(v.literal("small"), v.literal("medium"), v.literal("large")),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) {
      throw new ConvexError({
        message: "Usuario no autenticado",
        code: "UNAUTHENTICATED",
      });
    }

    const user = await ctx.db
      .query("users")
      .withIndex("by_token", (q) =>
        q.eq("tokenIdentifier", identity.tokenIdentifier),
      )
      .unique();

    if (!user) {
      throw new ConvexError({
        message: "Usuario no encontrado",
        code: "NOT_FOUND",
      });
    }

    // Check if NIT already exists
    const existingOrg = await ctx.db
      .query("organizations")
      .withIndex("by_nit", (q) => q.eq("nit", args.nit))
      .unique();

    if (existingOrg) {
      throw new ConvexError({
        message: "Ya existe una organización con este NIT",
        code: "CONFLICT",
      });
    }

    const organizationId = await ctx.db.insert("organizations", {
      name: args.name,
      nit: args.nit,
      address: args.address,
      phone: args.phone,
      email: args.email,
      industry: args.industry,
      size: args.size,
      createdById: user._id,
    });

    // Update user with organization
    await ctx.db.patch(user._id, {
      organizationId,
      role: "admin",
    });

    return await ctx.db.get(organizationId);
  },
});

export const get = query({
  args: {
    organizationId: v.id("organizations"),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) {
      throw new ConvexError({
        message: "Usuario no autenticado",
        code: "UNAUTHENTICATED",
      });
    }

    return await ctx.db.get(args.organizationId);
  },
});

export const getUserOrganization = query({
  args: {},
  handler: async (ctx) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) {
      return null;
    }

    const user = await ctx.db
      .query("users")
      .withIndex("by_token", (q) =>
        q.eq("tokenIdentifier", identity.tokenIdentifier),
      )
      .unique();

    if (!user || !user.organizationId) {
      return null;
    }

    return await ctx.db.get(user.organizationId);
  },
});

export const update = mutation({
  args: {
    organizationId: v.id("organizations"),
    name: v.optional(v.string()),
    address: v.optional(v.string()),
    phone: v.optional(v.string()),
    email: v.optional(v.string()),
    industry: v.optional(v.string()),
    size: v.optional(v.union(v.literal("small"), v.literal("medium"), v.literal("large"))),
  },
  handler: async (ctx, args) => {
    const identity = await ctx.auth.getUserIdentity();
    if (!identity) {
      throw new ConvexError({
        message: "Usuario no autenticado",
        code: "UNAUTHENTICATED",
      });
    }

    const user = await ctx.db
      .query("users")
      .withIndex("by_token", (q) =>
        q.eq("tokenIdentifier", identity.tokenIdentifier),
      )
      .unique();

    if (!user || user.role !== "admin") {
      throw new ConvexError({
        message: "No tiene permisos para realizar esta acción",
        code: "FORBIDDEN",
      });
    }

    const { organizationId, ...updates } = args;
    await ctx.db.patch(organizationId, updates);

    return await ctx.db.get(organizationId);
  },
});
